# 算法题

# 把数组排成最小值

输入一个非负整数数组，把数组里所有数字拼接起来排成一个数，打印能拼接出的所有数字中最小的一个。

**输入:** `[3,30,34,5,9]`
**输出:** "`3033459"`

思路：快排+自定义排序

```js
/**
 * @param {number[]} nums
 * @return {string}
 */
var minNumber = function(nums) {
    function swap(i,j){
        let temp = nums[i];
        nums[i] = nums[j];
        nums[j] = temp;
    }
    function compare(val1, val2){
        if(val1 === val2) return 0;
        let i = 0;
        while(i < val1.length && i < val2.length){
            if(val1[i] === val2[i]){
                i++;
            }else{
                if(val1[i] < val2[i]){
                    return 1;
                }else{
                    return -1;
                }
            }
        }
        if(i < val1.length) return -1;
        if(i < val2.length) return 1;
    }
    function quickSort(l, r){
		// 注意1：规定好区间[l,r)
        if(l >= r) return;
        let i = l, j = r - 1;
        let pos = nums[i];
        while(i < j){
	        // 注意2：j先行，占位，后面交换l和j
            while(i < j && compare(nums[j].toString() + pos.toString(), pos.toString() + nums[j].toString()) <= 0){
                j--;
            }
            while(i < j && compare(nums[i].toString() + pos.toString(), pos.toString() + nums[i].toString()) >= 0){
                i++;
            }
            if(i < j){
                swap(i,j);
            }
        }
        // 交换l和j，之后j位的值就出在正确的位置上。
        swap(l, j);
        // 排左右边
        quickSort(l,j);
        quickSort(j + 1,r);
    }
    quickSort(0,nums.length);
    let str = '';
    for(let c of nums){
        str += c.toString();
    }
    return str;
};
```
